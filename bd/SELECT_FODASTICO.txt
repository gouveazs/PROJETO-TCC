SELECT SUM(ITEM_PEDIDO.QUANTIDADE),PRODUTO.NOME 
	FROM PEDIDO JOIN ITEM_PEDIDO ON 
		(PEDIDO.COD_PEDIDO=ITEM_PEDIDO.COD_PEDIDO)
        JOIN PRODUTO ON (ITEM_PEDIDO.COD_PRODUTO=PRODUTO.COD_PRODUTO)
        WHERE PEDIDO.MES = 3
        GROUP BY PRODUTO.NOME
        ORDER BY ITEM_PEDIDO.QUANTIDADE DESC
